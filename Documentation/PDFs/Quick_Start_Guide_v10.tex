\section{Quick Start Workflow}

\subsection{Step 1: Obtain Source Data}

\textbf{Required Data Files} (see \texttt{sources/README.md} for details):
\begin{itemize}
    \item OCC historical call reports (1863-1947)
    \item FFIEC modern call reports (1959-2023)
    \item OCC receivership records
    \item FDIC failed bank data
    \item GFD macroeconomic data (CPI, yields, stocks)
    \item JST macroeconomic dataset
    \item FRED GDP data
\end{itemize}

Place all data files in \texttt{sources/} directory.

\subsection{Step 2: Install R Packages}

\begin{lstlisting}[language=R]
install.packages(c(
  "tidyverse", "haven", "fixest", "lubridate",
  "scales", "readxl", "here", "pROC",
  "sandwich", "lmtest"
))
\end{lstlisting}

\textbf{Note}: First run will auto-install any missing packages.

\subsection{Step 3: Run Complete Pipeline}

\begin{lstlisting}[language=R]
# Open RStudio, then:
source("code/00_master.R")
\end{lstlisting}

\section{Troubleshooting}

\subsection{``Cannot find sources/'' Error}
\textbf{Cause}: Source data files not obtained\\
\textbf{Fix}: See \texttt{sources/README.md} for download instructions

\subsection{``Package 'xxx' not available'' Error}
\textbf{Cause}: Missing R package\\
\textbf{Fix}: Run \texttt{install.packages("package\_name")}

\subsection{``Out of memory'' Error}
\textbf{Cause}: Insufficient RAM (< 16 GB)\\
\textbf{Fix}: Close other applications or upgrade RAM

\subsection{Scripts fail partway through}
\textbf{Cause}: Various (data issues, package conflicts)\\
\textbf{Fix}: Check \texttt{VALIDATION\_RUN\_LOG.md} for specific error message
